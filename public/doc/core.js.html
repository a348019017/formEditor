import { GUID } from "../../util";

//统一管理这些实例化的对象，通过guid进行索引
export const myobjects = {};

export const typeObject = {};

/**
 * 对象基类，处理object的管理等
 */
export class baseObject {
  /**
   * 对象基类，处理json配置的反序列化
   */
  constructor(element) {
    this.guid = GUID();
    if (element) {
      element.guid = this.guid;
    }
    myobjects[this.guid] = this;
  }

  destory() {
    myobjects[this.guid] = undefined;
  }

  //根据guid获取对象
  static getObjectByGUID(guid) {
    return myobjects[guid];
  }

  //全局注册对象
  static registerClass(type, cls) {
    typeObject[type] = cls;
  }

  static createObject(type, element, parent) {
    if (typeObject[type]) return new typeObject[type](element, parent);
  }
}
