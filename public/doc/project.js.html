import { Tile3DLayer } from "./layer";
import { baseObject } from "./core";
import { imageprovider } from "../../../index";

/**
 * 项目抽象类，管理一些通用配置，如UI widget，灯光全局配置等等
 * @extends baseObject
 */
export class Project extends baseObject {
  constructor() {
    super();
    //使用guid进行检索，guid不可更改，name不使用
    this._layershash = {};
  }
  get name() {
    return this._name;
  }
  set name(value) {
    this._name = value;
  }

  //
  get layers() {
    return Object.values(this._layershash);
  }

  add(element) {
    let lyr = baseObject.createObject(element.type, element, undefined);
    this._layershash[lyr._id] = lyr;
  }
  //移除指定图层
  remove(element) {
    let lyr = this._layershash[element.id];
    if (lyr) {
      lyr.destory();
      this._layershash[element.id] = undefined;
    }
  }

  get backgroundcolor() {
    return this._backgroundcolor;
  }

  //设置背景色
  set backgroundcolor(value) {
    return;
  }
  //设置imagelayer
  set imageprovider(value) {
    this._imageprovider = value;
    if (this._imageprovider) {
      imageprovider.addimagelayer(this._imageprovider);
    }
  }
  get imageprovider() {
    return this._imageprovider;
  }
}

//注册全局类，便于自主初始化对象
baseObject.registerClass("project", Project);
